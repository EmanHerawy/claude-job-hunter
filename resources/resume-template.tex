% --- Custom commands ---
\newcommand{\resumeheader}[5]{%
  % #1 Full name  #2 Location  #3 Phone  #4 Email  #5 LinkedIn/GitHub
  \begin{center}
    {\LARGE\bfseries #1}\\[4pt]
    #2 \textbullet\ #3 \textbullet\ \href{mailto:#4}{#4}\\[2pt]
    #5
  \end{center}
  \vspace{4pt}
}

\newcommand{\roleentry}[4]{%
  % #1 Company  #2 Title  #3 Location  #4 Dates
  \textbf{#1} \hfill \textit{#3}\\
  \textit{#2} \hfill #4
  \vspace{2pt}
}

\newcommand{\projectentry}[3]{%

  \textbf{#1} \textit{(#2)}%
  \ifx&#3&\else\ — \href{#3}{\texttt{#3}}\fi
}
